title Login
actor Administrator
boundary LoginView
control AuthController
entity UserEntity
Administrator->LoginView:login(email,password)
activate LoginView
LoginView -->LoginView : collectForm(data)
LoginView-->LoginView:validateForm(data)
alt validData = True
LoginView -->AuthController : requestToken(email,password)
activate AuthController
AuthController -->UserEntity : SearchByEmail(email)
activate UserEntity
UserEntity --> UserEntity : select(email) 
UserEntity -->AuthController : User
deactivate UserEntity
alt User!=None
AuthController -->AuthController : generateJWT(salt,data)
AuthController--> AuthController : prepareRequestBody(token)
AuthController -->LoginView : request200(token)
deactivate AuthController
LoginView -->LoginView : prepareRoutes()
LoginView ->Administrator : redirectsDashboard()
else 
AuthController --> LoginView: request400() 
LoginView ->Administrator : Displays Error
end
else 
LoginView->Administrator:displays(error)
end
